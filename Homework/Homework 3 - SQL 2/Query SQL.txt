/*
1. В каждом департаменте найти сотрудника, пришедшего на работу предпоследним по дате трудоустройства среди не уволенных на данный момент. Если в департаменте есть люди, пришедшие на работу в один и тот же (предпоследний) день, то выбрать их всех. Вывести id департамента, имена сотрудников и даты их приема
*/

select dep_id, name, hired_dt
from (select *, dense_rank() over (partition by dep_id order by hired_dt desc) hired_rank
      from employees
      where fired_dt is null)
where hired_rank = 2;

/*
2. Вывести, по каждому департаменту:
  - зарплату максимальную
  - минимальную
  - среднюю
  - отношение минимальной к максимальной, выраженное в процентах (результат округлить до двух знаков после запятой)
  - максимальное количество смен локаций, сделанных сотрудниками департамента
*/

select d.id as dep_id, d.name as dep_name, max(salary) max_salary, min(salary) min_salary, avg(salary) avg_salary, round(min(salary) * 100.0/max(salary), 2) ratio_salary, max(c_changes) max_changes
      from departments d
          left join employees n1 
          on d.id = n1.dep_id
          left join (select emp_id, count(emp_id) - 1 as c_changes
                            from emp_loc_history 
                            group by emp_id) n2
          on n1.id = n2.emp_id
      group by d.id, d.name
      order by d.id;

/*
3. Вывести общее, по всем департаментам, количество нанятых сотрудников в разбивке по годам (за 2020, 2021 годы и так далее). Уволенных нужно также взять в расчет. Ответ должен быть в формате Год, Количество сотрудников
*/

select extract(year from hired_dt) as year, count(*) emp_hired
      from employees
      group by extract(year from hired_dt)
      order by extract(year from hired_dt);

/*
4. Для каждого сотрудника вывести имя его коллеги по департаменту, устроившегося на работу сразу после него или в один с ним день. Для самого последнего устроившегося сотрудника отобразить прочерк '-'. Результат вывести в формате dep_id, Имя сотрудника, Имя следующего сотрудника (либо прочерк)
*/

select dep_id, name, next_name
from (select *, lead(name, 1, '-') over (partition by dep_id order by hired_dt) next_name
      from employees);

/*
5. Для каждого депаратамента вывести накопительную сумму выплачиваемых зарплат по мере поступления новых сотрудников на работу. Результат вывести в виде dep_id, hired_dt, накопительная сумма. Зарплаты уволенных сотрудников вычитать не нужно.
*/

select dep_id, hired_dt, sum_salary
from (select *, sum(salary) over (partition by dep_id order by hired_dt) sum_salary
      from employees);

/*
6. Только с помощью оконной функции last_value, для каждого департамента вывести сотрудника, имя которого - последнее в алфавитном порядке. Результат должен быть в виде dep_id, Имя сотрудника
*/

select distinct dep_id, last_name
from (select *, last_value(name) over (partition by dep_id order by name rows between unbounded preceding and unbounded following) last_name
      from employees)
order by dep_id;

/*
7. Используя только оконную функцию sum, для каждого сотрудника в департаменте вывести сумму зарплат его самого и двух коллег: одного пришедшего на работу до и одного - после в тот же департамент, если таковые нашлись. Если сотрудники устроились в один день, то любой из них может считаться предыдущим/следующим. Результат вывести в виде dep_id, name, hired_dt, сумма зарплат с сортировкой по dep_id и hired_dt
*/

select dep_id, name, hired_dt, sum_salary
from (select *, sum(salary) over (partition by dep_id order by hired_dt rows between 1 preceding and 1 following) sum_salary
      from employees);
