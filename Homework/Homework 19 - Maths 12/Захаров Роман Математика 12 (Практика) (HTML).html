import numpy as np
import pandas as pd
import statsmodels.api as sm
import matplotlib.pyplot as plt
import seaborn as sns
from scipy import stats

N = 1000
mu = [0, 0]
cov = [[1, -0.7],
       [-0.7, 1]]

np.random.seed(42)
X = np.random.multivariate_normal(mu, cov, N)
x1, x2 = X[:, 0], X[:, 1]

# Генерация y: y = beta0 + beta1*x1 + beta2*x2 + eps
beta0, beta1, beta2 = 5, 2.5, -1.5
eps = np.random.normal(0, 1, N)
y = beta0 + beta1 * x1 + beta2 * x2 + eps

df = pd.DataFrame({'y': y, 'x1': x1, 'x2': x2})

Оценка модели методом МНК и проверка остатков на нормальность¶

МНК очень точно восстановил заданные параметры. Коэффициенты $\hat{\beta}_1 \approx 2.5071$ и $\hat{\beta}_2 \approx -1.4628$ практически идентичны тем, что мы закладывали при генерации ($2.5$ и $-1.5$). Все p-values равны $0.000$. Это значит, что переменные $x_1$ и $x_2$ статистически значимо влияют на $y$. Качество ($R^2 = 0.929$) говорит о том, что модель объясняет $93\%$ изменчивости данных.

Проврека на нормальность остатков: Так как $p > 0.05$, мы принимаем гипотезу о том, что остатки распределены нормально.

Отрисовка графиков¶

По графикам y vs x1 и y vs x2 мы видим, что переменные x1 и x2 имеют, соотвественно, прямое и обратное влияние на y. Это соотвествует значениям коэффициентов ${\beta}_1 = 2.5$ (положительный) и ${\beta}_2 = -1.5$ (отрицательный)

График $y$ vs $\hat{y}$ - это главный индикатор качества модели. Чем ближе точки к диагонали, тем выше точность предсказания. Тот факт, что разброс минимален, соответствует высокому показателю $R^2 = 0.929$. Модель объясняет почти $93\%$ изменчивости целевой переменной, а значит, предсказанные значения $\hat{y}$ очень близки к реальным $y$.

Графики зависимости оценок коэффициентов $\hat{\beta}_1$ и $\hat{\beta}_2$ в зависимости от размера выборки $N$¶

При малых размерах выборки ($N < 100$) наблюдаются сильные колебания и широкие доверительные интервалы. С увеличением $N$ оценки $\hat{\beta}_1$ и $\hat{\beta}_2$ стабилизируются и стремятся к своим истинным значениям ($2.5$ и $-1.5$). Синяя область (95% CI) сужается по мере роста $N$, что наглядно демонстрирует закон больших чисел. Это означает, что точность нашего прогноза растет пропорционально объему собранных данных. После достижения размера выборки примерно в $600$ наблюдений оценки становятся стабильными, а их стандартные ошибки минимизируются.

results = []
for n in range(30, 1001, 10):
    X_n = X_model.iloc[:n]
    y_n = y[:n]
    res_n = sm.OLS(y_n, X_n).fit()

    results.append({
        'N': n,
        'b1': res_n.params['x1'],
        'b1_se': res_n.bse['x1'],
        'b2': res_n.params['x2'],
        'b2_se': res_n.bse['x2']
    })

res_df = pd.DataFrame(results)
res_df['b1_up'] = res_df['b1'] + 1.96 * res_df['b1_se']
res_df['b1_low'] = res_df['b1'] - 1.96 * res_df['b1_se']
res_df['b2_up'] = res_df['b2'] + 1.96 * res_df['b2_se']
res_df['b2_low'] = res_df['b2'] - 1.96 * res_df['b2_se']

# Графики
fig, ax = plt.subplots(1, 2, figsize=(15, 5))

for i, b in enumerate(['b1', 'b2']):
    true_val = beta1 if b == 'b1' else beta2
    ax[i].plot(res_df['N'], res_df[b], label='Оценка коэффициента', color='blue')
    ax[i].fill_between(res_df['N'], res_df[f'{b}_low'], res_df[f'{b}_up'], color='blue', alpha=0.2, label='95% CI')
    ax[i].axhline(true_val, color='red', linestyle='--', label='Истинное значение')
    ax[i].set_title(f'Сходимость {b} от N')
    ax[i].set_xlabel('Размер выборки N')
    ax[i].legend()

plt.show()
